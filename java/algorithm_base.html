public class Bubble {
    public static void main(String[] args) {
        // 需求: 从小到大比较
        int[] nums = { 1, 5, 4, 3, 2 };

        // 第一次排序
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                int temp = nums[i];
                nums[i] = nums[i + 1];
                nums[i + 1] = temp;
            }
        }

        for (int i = 0; i < nums.length; i++) {
            System.out.print(nums[i] + " ");
        }
        // 1 4 3 2 5 
    }
}

public class Bubble {

        // 需求: 从小到大比较

    }
}

  • 经验法

前提:

下标从 0 开始的

排序区和未排序区的划分, 开始时都是未排序区

趟数是否要考虑头尾元素 (for 循环 len() 是否要减 1, 考虑就不减 1)

口诀:

循环多少趟, 每趟走(比较)多少次

每次比较符合就替换(移动)

public class Bubble {
    public static void main(String[] args) {
        // 需求: 从小到大比较
        int[] nums = { 1, 3, 2, 5, 4 };
        // 排序区 未排序区 -> 排序区在后面(因为大的数会移到最后面)
        // 趟数是否考虑首尾元素(用两个做例子) -> 不考虑, 要减1
        // 循环多少趟, 走多少次 -> 循环 n-1 趟,每趟走 n-1-i 次
        // i的含义: 第几趟, 排序区个数
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = 0; j < nums.length - 1 - i; j++) {
                if (nums[j] > nums[j + 1]) {
                    int temp = nums[j];
                    nums[j] = nums[j + 1];
                    nums[j + 1] = temp;
                }
            }
        }

        for (int i = 0; i < nums.length; i++) {
            System.out.println(nums[i]);
        }
    }
}

# 二分查找

import java.util.Scanner;

public class BinarySearch {
    public static void main(String[] args) {
        int[] nums = { 1, 3, 5, 7, 19, 26 };

        Scanner scanner = new Scanner(System.in);
        System.out.println("请输入你要查找的数字:");
        int target = scanner.nextInt();

        // 需要两个变量来确定检索范围
        int left = 0;
        int right = nums.length - 1;

        if (target < nums[0] || target > nums[nums.length - 1]) {
            System.out.println("输入的数字不在范围内");
        } else {
            int tIndex = -1;
            // 左右边界不重合, 继续 // 边界是否独立的判断和处理
            while (left <= right) {
                int middle = (left + right) / 2;
                if (nums[middle] == target) {
                    tIndex = middle;
                    break;
                } else if (nums[middle] > target) {
                    right = middle - 1;
                } else {
                    left = middle + 1;
                }
            }

            System.out.println("索引是:" + tIndex);
        }
    }
}

# 算法复杂度

通常我们考虑一个算法的指标是看其在时间和空间两个维度的表现决定的

  • 时间维度: 是指执行算法消耗的时间, 通常用"时间复杂度"来描述
  • 空间维度: 是指执行算法占用的内存空间, 通常用"空间复杂度"来描述

最经典的方法: 大 O 表示法

# 时间复杂度

  • 线性阶

随着数据规模 n 增加, 计算的次数会线性增长

(计算次数 == 常数 * 数据规模 n)

  • 常数阶

无论数据规模 n 怎么增加, 计算的次数保持稳定不变, 和 n 无关

(计算次数 == 常数)

  • 指数阶

随着数据规模 n 增加, 计算的次数会指数级增长

(计算次数 == 数据规模 n 连续乘)

  • 对数阶

计算次数"连续乘某个数"的追赶达到数据规模 n , 这种增长是对数级别

数据规模 n"连续除某个数"的追赶达到计算次数 , 这种增长是对数级别

(计算次数 * 某个数 == 数据规模 n)

(计算次数 == 数据规模 n / 某个数)

package com.baicai;

public class Test {

  public static void test1(int n) {
    // 1
    if (n > 10) { 
      System.out.println("n > 10");
    } else if (n > 5) { // 2
      System.out.println("n > 5");
    } else {
      System.out.println("n <= 5"); 
    }


    // 1 + 4 + 4 + 4
    for (int i = 0; i < 4; i++) {
      System.out.println("test");
    }

    // 14
    // O(1)
  }

  public static void test2(int n) {
    // 1 + 3n
    for (int i = 0; i < n; i++) {
      System.out.println("test");
    }

    // O(n)
  }

  public static void test3(int n) {
    // 1 + 2n + n * (1 + 3n)
    // 1 + 2n + n + 3n^2
    // 3n^2 + 3n + 1
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        System.out.println("test");
      }
    }

    // O(n^2)
  }

  public static void test4(int n) {
    // 1 + 2n + n * (1 + 45)
    // 1 + 2n + 46n
    // 48n + 1
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < 15; j++) {
        System.out.println("test");
      }
    }

    // O(n)
  }

  public static void test5(int n) {
    // 8 = 2^3
    // 16 = 2^4

    // 3 = log2(8)
    // 4 = log2(16)

    // 执行次数 = log2(n)

    while ((n = n / 2) > 0) { // 连续除以2 --> 就是求n能除多少次2 
      System.out.println("test");
    }

    // O(logn)
  }

  public static void test6(int n) {
    // log5(n)
    while ((n = n / 5) > 0) { // 连续除以5 --> 就是求n能除多少次5 
      System.out.println("test");
    }

    // O(logn)
  }

  public static void test7(int n) {
    // 1 + 2*log2(n) + log2(n) * (1 + 3n)
    // 1 + 3*log2(n) + 2 * nlog2(n)

    for (int i = 1; i < n; i = i * 2) { //就是求i能乘多少次2

      // 1 + 3n
      for (int j = 0; j < n; j++) {
        System.out.println("test");
      }
    }

    //O(nlogn)
  }

}

# 归并思路

类似"两个队打擂台"

  • 同时遍历两个数组, 遍历的当前值进行比较, 小的放入新数组

  • 那个数组的被放入, 就指针后移, 继续遍历

  • 依此类推, 多次遍历比较后

public class Merge {
    public static void main(String[] args) {
        int[] arr1 = { 1, 3, 5, 8, 12, 35, 54, 456 };
        int[] arr2 = { 2, 4, 7, 10, 40, 60 };

        // 代表上擂台的人的下标
        int i = 0;
        int j = 0;

        // 创建一个大数组
        int[] newArr = new int[arr1.length + arr2.length];

        // 开始一一打擂台
        while (i < arr1.length || j < arr2.length) {
            // arr1比完, arr2剩余的放入
            if (i == arr1.length) {
                newArr[i + j] = arr2[j];
                j++;
                continue;
            }
            // arr2比完, arr1剩余的放入
            if (j == arr2.length) {
                newArr[i + j] = arr1[i];
                i++;
                continue;
            }

            // 谁小 谁放入新数组
            if (arr1[i] <= arr2[j]) {
                newArr[i + j] = arr1[i];
                i++;
            } else {
                newArr[i + j] = arr2[j];
                j++;
            }
        }

        for (int k = 0; k < newArr.length; k++) {
            System.out.print(newArr[k]+" ");
        }
    }
}
