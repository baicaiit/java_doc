# 初始化

Calendar 类提供了对“年月日时分秒星期”等日期信息操作的函数，并针对不同国家和地区的日历提供了相应的子类，即本地化, 比如公历 GregorianCalendar，佛历 BuddhistCalendar 等, 一般不用自己使用子类

# Instant

时间线上的瞬间点

Instant可以代替Date类，那它肯定可以获取当前时间

package api.date;

import java.time.Instant;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.TimeZone;

public class InstantTest {
    public static void main(String[] args) {
        // 获取当前时间点
        Instant instant = Instant.now();
        System.out.println(instant); // 2023-02-22T10:05:19.445303Z
        // 世界统一时间UTC,  比北京时间少了8个小时

        System.out.println(ZoneId.systemDefault()); // Asia/Shanghai
        System.out.println(TimeZone.getDefault()); // ZoneInfo[id="Asia/Shanghai",offset=28800000,dstSavings=0,useDaylight=false,transitions=31,lastRule=null]

        // 输出北京时间
        ZonedDateTime atZone = instant.atZone(ZoneId.of("Asia/Shanghai"));
        System.out.println(atZone); // 2023-02-22T18:35:42.413259+08:00[Asia/Shanghai]

        // 获取时间戳
        System.out.println(instant.getEpochSecond()); // 秒
        System.out.println(instant.toEpochMilli()); // 毫秒

        System.out.println(atZone.toEpochSecond()); // 秒
        Instant instant2 = atZone.toInstant(); // 转换成Instant

        System.out.println(System.currentTimeMillis()); //毫秒
    }
}

import java.time.Instant;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.TimeZone;

public class InstantTest {
    public static void main(String[] args) {
        // 将时间戳转换成Instant
        //// 秒级
        System.out.println(Instant.ofEpochSecond(1677062379)); // 2023-02-22T10:39:39Z
        //// 毫秒级
        System.out.println(Instant.ofEpochMilli(1677062379477L)); // 2023-02-22T10:39:39.477Z

        Instant instant = Instant.now();
        System.out.println(Instant.ofEpochSecond(1677062379, instant.getNano())); // 2023-02-22T10:39:39.845268Z

        // 将表达式转换为Instant
        String text = "2020-06-10T08:46:55.967Z";
        Instant parseInstant = Instant.parse(text);
    }
}

import java.time.LocalDate;
import java.time.Month;

public class LocalDateTimeTest {
    public static void main(String[] args) {
        // 指定日期转换成LocalDate
        LocalDate birthday = LocalDate.of(1991, 7, 16);
        System.out.println("birthday: " + birthday);

        // 比较日期是否相等
        LocalDate localDate1 = LocalDate.now();
        LocalDate localDate2 = LocalDate.of(1991, 7, 16);
        if (localDate1.equals(localDate2)) {
            System.out.println("localDate1 equals localDate2");
        }

        // 获取日期是本周/本月/本年的第几天
        LocalDate today = LocalDate.now();
        System.out.println("Today:" + today);
        System.out.println("Today is:" + today.getDayOfWeek());
        System.out.println("今天是本周的第" + today.getDayOfWeek().getValue() + "天");
        System.out.println("今天是本月的第" + today.getDayOfMonth() + "天");
        System.out.println("今天是本年的第" + today.getDayOfYear() + "天");

        // 判断是否为闰年
        System.out.println(today.getYear() + " is leap year:" + today.isLeapYear());
    }
}

# BigDecimal 类

不能用 double 表示钱:

因为浮点数不能准确代表我们用于货币的基数10的倍数

在商业计算中要用java.math.BigDecimal

BigDecimal 所创建的是对象，我们不能使用传统的+、-、*、/等算术运算符直接对其对象进行数学运算，而必须调用其相对应的方法

方法中的参数也必须是 BigDecimal 的对象

本质是记录一个精度信息，使用整数进行计算

因为十进制整数在转化成二进制数时不会有精度问题，那么把十进制小数扩大N倍让它在整数的维度上进行计算，并保留相应的精度信息

import java.math.BigDecimal;

public class BigDecimalTest {
    public static void main(String[] args) {
      	BigDecimal bigDecimal1 = new BigDecimal(0.1);
        BigDecimal bigDecimal2 = new BigDecimal(0.2);
      
        BigDecimal add = bigDecimal1.add(bigDecimal2);
      
        double doubleValue = add.doubleValue();
        System.out.println(doubleValue);
    }
}

# Arrays 类

该类包含用于操作数组的各种方法（如排序和搜索）

package api.tool;

import java.util.Arrays;
import java.util.Random;

public class ArraysTest {
    public static void main(String[] args) {
        byte[] nums = new byte[5];
        Random random = new Random(1);
        random.nextBytes(nums);

        // 更友好的输出数组
        System.out.println(Arrays.toString(nums));

        // 排序
        Arrays.sort(nums);
        System.out.println(Arrays.toString(nums));

        Arrays.parallelSort(nums);
        System.out.println(Arrays.toString(nums));

        // 二分查找
        int i = Arrays.binarySearch(nums, (byte) 115);
        System.out.println(i);

        // 拷贝
        byte[] nums1 = Arrays.copyOf(nums, 10);
        System.out.println(Arrays.toString(nums1));

        byte[] nums2 = Arrays.copyOfRange(nums, 0, 2);
        System.out.println(Arrays.toString(nums2));

        // 比较
        System.out.println(Arrays.equals(nums1, nums2));
    }
}
